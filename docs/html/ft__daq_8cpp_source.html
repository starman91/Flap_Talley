158 void FTDAQ::storeTargetAmplitudePoints(std::vector<long double>& triggerData, std::vector<long double>& rawData)
159 {
160  //Monitor testing
161  //QFile file("flapping_found_amplitudes.csv");
162  // if (!file.open(QIODevice::Append| QIODevice::Text))
163  // return;
164  //QTextStream out(&file);
165 
166  QTime mark;
167  //Plottables
168  QList<QPointF> targetPoints;
169 
170  //Store raw data points at this time slice
171  // TODO: Set up for multiple targets
172  v_targets.at(0).setSignal(rawData);
173 
174  // Init tracking variables
175  int rise = -1;
176  int fall = -1;
177  for (int i = 0; i < triggerData.size(); i++)
178  {
179  if ((0 < i) && (i < triggerData.size() - 1))
180  {
181  // Check for rising edge of signal
182  if ((triggerData.at(i) == 0) && (triggerData.at(i + 1) == 1))
183  {
184  rise = i;
185  }
186  //Find falling edge of ID signal
187  else if ((triggerData.at(i) == 1) && (triggerData.at(i + 1) == 0))
188  {
189  fall = i;
190  }
191  //Examine whole trigger
192  if ((rise > 0) && (fall > 0))
193  {
194  int targetX = 0;
195  (fall - rise != 1) ? targetX = fall - (fall - rise) / 2 : targetX = rise;
196 
197  // Set plot point of center of peak.
198  QPointF centerPoint;
199  centerPoint.setX(double(targetX + 1) / 10.0);
200  centerPoint.setY(rawData.at(targetX + 1));
201  targetPoints.push_back(centerPoint);
202 
203  // Find maximum value
204  QList<long double> signal;
205  for (int point = rise - 5; point < fall + 5; point++)
206  {
207  signal.append(rawData.at(point));
208  }
209  //target.setSignal(signal);
210  long double max = *std::max_element(signal.begin(), signal.end());
211  // TODO: Expand for multiple targets
212  v_targets.at(0).setCurrentAmplitude(max); // tracking point
213  QPointF targetPoint;
214  targetPoint.setX (QTime::currentTime().msecsSinceStartOfDay()/1000.0);
215  targetPoint.setY ( g_status.getSimTargetSize() - max);
216  g_support.addTargetData(targetPoint);
217  //if (g_status.getSimFlap())
218  //out << QTime::currentTime().msecsSinceStartOfDay() << ", " << double(max) << "\n";
219 
220  // Reset tracking variables
221  rise = -1;
222  fall = -1;
223  }
224  }
225  }
226  return;
227 }
