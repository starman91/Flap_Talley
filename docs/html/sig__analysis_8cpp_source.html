168

179  plan_forward = fftw_plan_dft_r2c_1d ( N, in, out, FFTW_ESTIMATE );
180  fftw_execute ( plan_forward );
181 
182  //Calculate the y axis of the power spectrum in dB
183  std::vector<double>v(N);
184  for (int i = 0; i<= ((N/2)-1); i++)
185  {

187  }

203  //Clean up
