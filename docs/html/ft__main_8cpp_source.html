175  radarChart->addSeries(trackingData);
176  trackingData->attachAxis(radarChart->axisY());
177  trackingData->attachAxis(radarChart->axisX());
178 
179  // Marks maximum value point found during any target discovery
180  QScatterSeries*targetingData = new QScatterSeries();
181  targetingData->setColor("red");
182  targetingData->setMarkerSize(5);
183  targetingData->append(g_support.getTargetPointData());
184  radarChart->addSeries(targetingData);
185  targetingData->attachAxis(radarChart->axisY());
186  targetingData->attachAxis(radarChart->axisX());
187  }
188 
189  radarChart->createDefaultAxes();
190  radarChart->axisX()->setRange(0, 100);
191  radarChart->axisY()->setRange(-1, 10);
192  g_support.clearInData(); //Clear the data bucket
193  sem_recdata.release(1);
194  }
195  }
196 
197  if (g_status.getCalcFlap())
198  main_ui.label_found_flap_rate->setText(QString::number(g_status.getFoundFlapRate(), 'f', 3));
